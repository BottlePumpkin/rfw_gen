# rfw_gen v0.1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the core conversion engine and build_runner generator that converts `@RfwWidget`-annotated Flutter functions into rfwtxt/binary, supporting 5 core widgets (Text, Column, Row, Container, SizedBox).

**Architecture:** Monorepo with two packages — `rfw_gen` (core engine: annotations, AST→rfwtxt converter, widget registry) and `rfw_gen_builder` (custom `Builder` for build_runner). The core engine uses `package:analyzer` for AST parsing and `package:rfw` for rfwtxt validation/binary encoding. All widget mappings live in a `WidgetRegistry` that maps Flutter widget constructors to their rfwtxt equivalents.

**Tech Stack:** Dart 3.x, Flutter 3.x, `package:analyzer`, `package:rfw`, `package:build`, `package:build_runner`, `package:melos`, `package:test`, `package:flutter_test`

**Spec:** `docs/superpowers/specs/2026-03-23-rfw-gen-design.md`

// packages/rfw_gen/lib/src/errors.dart

/// Severity level for conversion issues.
enum RfwGenSeverity { fatal, warning }

/// An issue found during conversion.
class RfwGenIssue {
  final RfwGenSeverity severity;
  final String message;
  final String? suggestion;
  final int? line;

  const RfwGenIssue({
    required this.severity,
    required this.message,
    this.suggestion,
    this.line,
  });

  bool get isFatal => severity == RfwGenSeverity.fatal;

  @override
  String toString() {
    final buffer = StringBuffer('[rfw_gen] ');
    buffer.write(isFatal ? 'Error' : 'Warning');
    if (line != null) buffer.write(' (line $line)');
    buffer.write(': $message');
    if (suggestion != null) buffer.write('\n  Suggestion: $suggestion');
    return buffer.toString();
  }
}

/// Exception thrown when conversion encounters fatal errors.
class RfwGenException implements Exception {
  final List<RfwGenIssue> issues;

  const RfwGenException(this.issues);

  @override
  String toString() => issues.map((i) => i.toString()).join('\n');
}
```

/// Base class for all IR values.
sealed class IrValue {}

/// A string literal: `"hello"`.
class IrStringValue extends IrValue {
  final String value;
  IrStringValue(this.value);
}

/// A floating-point number: `24.0`.
class IrNumberValue extends IrValue {
  final double value;
  IrNumberValue(this.value);
}

/// An integer (used for colors, icon codes): `0xFF000000`.
class IrIntValue extends IrValue {
  final int value;
  IrIntValue(this.value);
}

/// A boolean: `true` / `false`.
class IrBoolValue extends IrValue {
  final bool value;
  IrBoolValue(this.value);
}

/// An enum string: `"center"`, `"start"`.
class IrEnumValue extends IrValue {
  final String value;
  IrEnumValue(this.value);
}

/// A list of values: `[16.0, 8.0]`.
class IrListValue extends IrValue {
  final List<IrValue> values;
  IrListValue(this.values);
}

/// A map of values: `{ fontSize: 24.0 }`.
class IrMapValue extends IrValue {
  final Map<String, IrValue> entries;
  IrMapValue(this.entries);
}

/// A widget node in the IR tree.
class IrWidgetNode extends IrValue {
  final String name;
  final Map<String, IrValue> properties;

  IrWidgetNode({
    required this.name,
    this.properties = const {},
  });
}

// packages/rfw_gen/lib/src/widget_registry.dart

/// How a widget accepts children.
enum ChildType { none, child, optionalChild, childList }

/// How a Dart parameter maps to an rfwtxt parameter.
class ParamMapping {
  /// The rfwtxt parameter name.
  final String rfwName;

  /// Optional transformer key (e.g., 'color', 'edgeInsets', 'textStyle').
  final String? transformer;

  const ParamMapping(this.rfwName, {this.transformer});

  /// Direct pass-through (same name, no transformation).
  const ParamMapping.direct(String name) : rfwName = name, transformer = null;
}

/// Defines how a Flutter widget maps to rfwtxt.
class WidgetMapping {
  /// The widget name in rfwtxt output.
  final String rfwName;

  /// Named parameter mappings: Dart param name → rfwtxt mapping.
  final Map<String, ParamMapping> params;

  /// If non-null, the first positional argument maps to this rfwtxt param.
  final String? positionalParam;

  /// How this widget accepts children.
  final ChildType childType;

  /// The parameter name for children (e.g., 'children', 'child').
  final String? childParam;

  /// Required rfwtxt import (e.g., 'core.widgets', 'material').
  final String import;

  const WidgetMapping({
    required this.rfwName,
    this.params = const {},
    this.positionalParam,
    this.childType = ChildType.none,
    this.childParam,
    this.import = 'core.widgets',
  });
}

/// Registry of supported widget mappings.
class WidgetRegistry {
  final Map<String, WidgetMapping> _widgets = {};

  /// All registered widget mappings.
  Map<String, WidgetMapping> get supportedWidgets =>
      Map.unmodifiable(_widgets);

  /// Check if a widget is supported.
  bool isSupported(String widgetName) => _widgets.containsKey(widgetName);

  /// Register a custom widget mapping.
  void register(String name, WidgetMapping mapping) {
    _widgets[name] = mapping;
  }

  /// Creates a registry with v0.1 core widgets.
  factory WidgetRegistry.core() {
    final registry = WidgetRegistry();

    registry.register('Text', WidgetMapping(
      rfwName: 'Text',
      positionalParam: 'text',
      params: {
        'style': ParamMapping('style', transformer: 'textStyle'),
        'textAlign': ParamMapping('textAlign', transformer: 'enum'),
        'maxLines': ParamMapping.direct('maxLines'),
        'overflow': ParamMapping('overflow', transformer: 'enum'),
        'softWrap': ParamMapping.direct('softWrap'),
      },
    ));

    registry.register('Column', WidgetMapping(
      rfwName: 'Column',
      childType: ChildType.childList,
      childParam: 'children',
      params: {
        'mainAxisAlignment': ParamMapping('mainAxisAlignment', transformer: 'enum'),
        'mainAxisSize': ParamMapping('mainAxisSize', transformer: 'enum'),
        'crossAxisAlignment': ParamMapping('crossAxisAlignment', transformer: 'enum'),
        'verticalDirection': ParamMapping('verticalDirection', transformer: 'enum'),
      },
    ));

    registry.register('Row', WidgetMapping(
      rfwName: 'Row',
      childType: ChildType.childList,
      childParam: 'children',
      params: {
        'mainAxisAlignment': ParamMapping('mainAxisAlignment', transformer: 'enum'),
        'mainAxisSize': ParamMapping('mainAxisSize', transformer: 'enum'),
        'crossAxisAlignment': ParamMapping('crossAxisAlignment', transformer: 'enum'),
        'verticalDirection': ParamMapping('verticalDirection', transformer: 'enum'),
      },
    ));

    registry.register('Container', WidgetMapping(
      rfwName: 'Container',
      childType: ChildType.optionalChild,
      childParam: 'child',
      params: {
        'color': ParamMapping('color', transformer: 'color'),
        'padding': ParamMapping('padding', transformer: 'edgeInsets'),
        'margin': ParamMapping('margin', transformer: 'edgeInsets'),
        'width': ParamMapping.direct('width'),
        'height': ParamMapping.direct('height'),
        'alignment': ParamMapping('alignment', transformer: 'alignment'),
        'decoration': ParamMapping('decoration', transformer: 'boxDecoration'),
      },
    ));

    registry.register('SizedBox', WidgetMapping(
      rfwName: 'SizedBox',
      childType: ChildType.optionalChild,
      childParam: 'child',
      params: {
        'width': ParamMapping.direct('width'),
        'height': ParamMapping.direct('height'),
      },
    ));

    return registry;
  }
}

/// Helper to parse a Dart expression string into an AST Expression node.
Expression parseExpression(String code) {
  final result = parseString(content: 'final x = $code;');
  final unit = result.unit;
  final decl = unit.declarations.first as TopLevelVariableDeclaration;
  return decl.variables.variables.first.initializer!;
}

void main() {

  setUp(() {
    converter = ExpressionConverter();
  });

  group('literals', () {
    test('string literal', () {
      final expr = parseExpression("'hello'");

      expect((result as IrStringValue).value, 'hello');

      final expr = parseExpression('24.0');

import 'package:analyzer/dart/ast/ast.dart';

import 'ir.dart';

/// Error thrown when an expression cannot be converted to rfwtxt.
class UnsupportedExpressionError implements Exception {
  final String message;
  final int? offset;
  UnsupportedExpressionError(this.message, {this.offset});

  @override
  String toString() => 'UnsupportedExpressionError: $message';
}

/// Converts Dart AST expressions to IR values.
class ExpressionConverter {
  /// Convert an AST Expression to an IrValue.
  IrValue convert(Expression expr) {
    return switch (expr) {
      StringLiteral() => _convertString(expr),
      IntegerLiteral() => IrIntValue(expr.value ?? 0),
      DoubleLiteral() => IrNumberValue(expr.value),
      BooleanLiteral() => IrBoolValue(expr.value),
      ListLiteral() => _convertList(expr),
      SetOrMapLiteral() => _convertMap(expr as SetOrMapLiteral),
      InstanceCreationExpression() => _convertConstructor(expr),
      PrefixedIdentifier() => _convertPrefixed(expr),
      MethodInvocation() => _convertMethodInvocation(expr),
      _ => throw UnsupportedExpressionError(
          'Expression type ${expr.runtimeType} is not supported',
          offset: expr.offset,
        ),
    };
  }

  IrValue _convertString(StringLiteral expr) {
    return IrStringValue(expr.stringValue ?? '');
  }

  IrValue _convertList(ListLiteral expr) {
    return IrListValue(expr.elements.map((e) {
      if (e is Expression) return convert(e);
      throw UnsupportedExpressionError('Non-expression list element');
    }).toList());
  }

  IrValue _convertMap(SetOrMapLiteral expr) {
    final entries = <String, IrValue>{};
    for (final element in expr.elements) {
      if (element is MapLiteralEntry) {
        final key = element.key;
        if (key is SimpleStringLiteral) {
          entries[key.value] = convert(element.value);
        }
      }
    }
    return IrMapValue(entries);
  }

  IrValue _convertConstructor(InstanceCreationExpression expr) {
    final name = expr.constructorName.type.name2.lexeme;
    final constructorName = expr.constructorName.name?.name;

    return switch (name) {
      'Color' => _convertColor(expr),
      'EdgeInsets' => _convertEdgeInsets(expr, constructorName),
      _ => throw UnsupportedExpressionError(
          'Constructor $name is not a known type',
          offset: expr.offset,
        ),
    };
  }

  IrValue _convertColor(InstanceCreationExpression expr) {
    final args = expr.argumentList.arguments;
    if (args.isEmpty) {
      throw UnsupportedExpressionError('Color requires a value argument');
    }
    final first = args.first;
    if (first is IntegerLiteral) {
      return IrIntValue(first.value ?? 0);
    }
    throw UnsupportedExpressionError('Color argument must be an integer literal');
  }

  IrValue _convertEdgeInsets(InstanceCreationExpression expr, String? constructor) {
    final args = expr.argumentList.arguments;

    return switch (constructor) {
      'all' => _edgeInsetsAll(args),
      'symmetric' => _edgeInsetsSymmetric(args),
      'only' => _edgeInsetsOnly(args),
      'fromLTRB' => _edgeInsetsFromLTRB(args),
      _ => throw UnsupportedExpressionError('EdgeInsets.$constructor is not supported'),
    };
  }

  IrValue _edgeInsetsAll(NodeList<Expression> args) {
    if (args.isEmpty) throw UnsupportedExpressionError('EdgeInsets.all requires a value');
    final value = _extractDouble(args.first);
    return IrListValue([IrNumberValue(value)]);
  }

  IrValue _edgeInsetsSymmetric(NodeList<Expression> args) {
    double horizontal = 0.0;
    double vertical = 0.0;
    for (final arg in args) {
      if (arg is NamedExpression) {
        final name = arg.name.label.name;
        final value = _extractDouble(arg.expression);
        if (name == 'horizontal') horizontal = value;
        if (name == 'vertical') vertical = value;
      }
    }
    return IrListValue([
      IrNumberValue(horizontal),
      IrNumberValue(vertical),
      IrNumberValue(horizontal),
      IrNumberValue(vertical),
    ]);
  }

  IrValue _edgeInsetsOnly(NodeList<Expression> args) {
    double left = 0.0, top = 0.0, right = 0.0, bottom = 0.0;
    for (final arg in args) {
      if (arg is NamedExpression) {
        final name = arg.name.label.name;
        final value = _extractDouble(arg.expression);
        switch (name) {
          case 'left': left = value;
          case 'top': top = value;
          case 'right': right = value;
          case 'bottom': bottom = value;
        }
      }
    }
    return IrListValue([
      IrNumberValue(left), IrNumberValue(top),
      IrNumberValue(right), IrNumberValue(bottom),
    ]);
  }

  IrValue _edgeInsetsFromLTRB(NodeList<Expression> args) {
    if (args.length < 4) {
      throw UnsupportedExpressionError('EdgeInsets.fromLTRB requires 4 arguments');
    }
    return IrListValue(args.take(4).map((a) => IrNumberValue(_extractDouble(a))).toList());
  }

  /// PrefixedIdentifier: e.g., MainAxisAlignment.center → enum value.
  IrValue _convertPrefixed(PrefixedIdentifier expr) {
    final prefix = expr.prefix.name;
    final property = expr.identifier.name;

    // Known enum types
    const knownEnums = {
      'MainAxisAlignment', 'CrossAxisAlignment', 'MainAxisSize',
      'VerticalDirection', 'TextAlign', 'TextOverflow', 'TextDirection',
      'Axis', 'Clip', 'BoxFit', 'StackFit', 'FlexFit',
      'WrapAlignment', 'WrapCrossAlignment',
    };

    if (knownEnums.contains(prefix)) {
      return IrEnumValue(property);
    }

    throw UnsupportedExpressionError(
      '$prefix.$property is not a recognized enum or constant',
      offset: expr.offset,
    );
  }

  /// MethodInvocation: handles Color(), EdgeInsets.all() etc.
  /// Note: parseString without imports parses these as MethodInvocation,
  /// not InstanceCreationExpression.
  IrValue _convertMethodInvocation(MethodInvocation expr) {
    final target = expr.target;

    // Static method: EdgeInsets.all(16)
    if (target is SimpleIdentifier) {
      final typeName = target.name;
      final methodName = expr.methodName.name;
      if (typeName == 'EdgeInsets') {
        return _convertEdgeInsetsArgs(methodName, expr.argumentList.arguments);
      }
    }

    // Free function style: Color(0xFF000000), Text('hello')
    if (target == null) {
      final name = expr.methodName.name;
      if (name == 'Color') {
        return _convertColorArgs(expr.argumentList.arguments);
      }
    }

    throw UnsupportedExpressionError(
      'Method call ${expr.methodName.name} is not supported',
      offset: expr.offset,
    );
  }

  IrValue _convertEdgeInsets(InstanceCreationExpression expr, String? constructor) {
    return _convertEdgeInsetsArgs(constructor, expr.argumentList.arguments);
  }

  IrValue _convertEdgeInsetsArgs(String? constructor, NodeList<Expression> args) {
    return switch (constructor) {
      'all' => _edgeInsetsAll(args),
      'symmetric' => _edgeInsetsSymmetric(args),
      'only' => _edgeInsetsOnly(args),
      'fromLTRB' => _edgeInsetsFromLTRB(args),
      _ => throw UnsupportedExpressionError('EdgeInsets.$constructor is not supported'),
    };
  }

  IrValue _convertColorArgs(NodeList<Expression> args) {
    if (args.isEmpty) {
      throw UnsupportedExpressionError('Color requires a value argument');
    }
    final first = args.first;
    if (first is IntegerLiteral) {
      return IrIntValue(first.value ?? 0);
    }
    throw UnsupportedExpressionError('Color argument must be an integer literal');
  }

  double _extractDouble(Expression expr) {
    if (expr is DoubleLiteral) return expr.value;
    if (expr is IntegerLiteral) return (expr.value ?? 0).toDouble();
    throw UnsupportedExpressionError('Expected numeric literal, got ${expr.runtimeType}');
  }
}

// packages/rfw_gen/lib/src/ast_visitor.dart
import 'package:analyzer/dart/ast/ast.dart';

import 'expression_converter.dart';
import 'ir.dart';
import 'widget_registry.dart';

/// Extracts an IR widget tree from a function's AST.
class WidgetAstVisitor {
  final WidgetRegistry registry;
  final ExpressionConverter expressionConverter;

  WidgetAstVisitor({
    required this.registry,
    required this.expressionConverter,
  });

  /// Extract the widget tree from a function declaration's return statement.
  IrWidgetNode extractWidgetTree(FunctionDeclaration function) {
    final body = function.functionExpression.body;
    Expression? returnExpr;

    if (body is ExpressionFunctionBody) {
      returnExpr = body.expression;
    } else if (body is BlockFunctionBody) {
      for (final statement in body.block.statements) {
        if (statement is ReturnStatement) {
          returnExpr = statement.expression;
          break;
        }
      }
    }

    if (returnExpr == null) {
      throw UnsupportedExpressionError('Function has no return statement');
    }

    return _convertWidget(returnExpr);
  }

  /// Convert an expression that should be a widget constructor call.
  IrWidgetNode _convertWidget(Expression expr) {
    if (expr is! InstanceCreationExpression) {
      // Try as a simple identifier constructor (e.g., Text('hello'))
      if (expr is MethodInvocation && expr.target == null) {
        return _convertFunctionStyleWidget(expr);
      }
      throw UnsupportedExpressionError(
        'Expected widget constructor, got ${expr.runtimeType}',
        offset: expr.offset,
      );
    }

    final name = expr.constructorName.type.name2.lexeme;
    if (!registry.isSupported(name)) {
      throw UnsupportedExpressionError(
        '`$name` is not supported by RFW',
        offset: expr.offset,
      );
    }

    final mapping = registry.supportedWidgets[name]!;
    final properties = <String, IrValue>{};

    for (final arg in expr.argumentList.arguments) {
      if (arg is NamedExpression) {
        _processNamedArg(arg, mapping, properties);
      } else {
        // Positional argument
        if (mapping.positionalParam != null) {
          properties[mapping.positionalParam!] =
              expressionConverter.convert(arg);
        }
      }
    }

    return IrWidgetNode(name: mapping.rfwName, properties: properties);
  }

  /// Handle `Text('hello')` style where analyzer parses it as MethodInvocation.
  IrWidgetNode _convertFunctionStyleWidget(MethodInvocation expr) {
    final name = expr.methodName.name;
    if (!registry.isSupported(name)) {
      throw UnsupportedExpressionError(
        '`$name` is not supported by RFW',
        offset: expr.offset,
      );
    }

    final mapping = registry.supportedWidgets[name]!;
    final properties = <String, IrValue>{};

    for (final arg in expr.argumentList.arguments) {
      if (arg is NamedExpression) {
        _processNamedArg(arg, mapping, properties);
      } else {
        if (mapping.positionalParam != null) {
          properties[mapping.positionalParam!] =
              expressionConverter.convert(arg);
        }
      }
    }

    return IrWidgetNode(name: mapping.rfwName, properties: properties);
  }

  void _processNamedArg(
    NamedExpression arg,
    WidgetMapping mapping,
    Map<String, IrValue> properties,
  ) {
    final paramName = arg.name.label.name;

    // Check if this is the child/children parameter
    if (paramName == mapping.childParam) {
      switch (mapping.childType) {
        case ChildType.child:
        case ChildType.optionalChild:
          properties['child'] = _convertWidget(arg.expression);
        case ChildType.childList:
          properties['children'] = _convertChildList(arg.expression);
        case ChildType.none:
          break;
      }
      return;
    }

    // Check if this is a known parameter
    if (mapping.params.containsKey(paramName)) {
      final paramMapping = mapping.params[paramName]!;
      properties[paramMapping.rfwName] =
          expressionConverter.convert(arg.expression);
      return;
    }

    // Unknown parameter — try to convert anyway (may be passed through)
    try {
      properties[paramName] = expressionConverter.convert(arg.expression);
    } on UnsupportedExpressionError {
      // Silently skip unsupported optional parameters
    }
  }

  IrListValue _convertChildList(Expression expr) {
    if (expr is! ListLiteral) {
      throw UnsupportedExpressionError('children must be a list literal');
    }
    return IrListValue(
      expr.elements.map((e) {
        if (e is Expression) return _convertWidget(e);
        throw UnsupportedExpressionError('Non-expression child');
      }).toList(),
    );
  }
}

// packages/rfw_gen/lib/src/rfwtxt_emitter.dart
import 'ir.dart';

/// Converts IR widget trees to rfwtxt string format.
class RfwtxtEmitter {
  /// Emit a complete rfwtxt file with imports and widget declaration.
  String emit({
    required String widgetName,
    required IrWidgetNode root,
    required Set<String> imports,
  }) {
    final buffer = StringBuffer();

    // Imports
    for (final import in imports.toList()..sort()) {
      buffer.writeln('import $import;');
    }
    buffer.writeln();

    // Widget declaration
    buffer.write('widget $widgetName = ');
    _emitWidget(buffer, root, indent: 0);
    buffer.writeln(';');

    return buffer.toString();
  }

  void _emitWidget(StringBuffer buffer, IrWidgetNode node, {required int indent}) {
    buffer.write('${node.name}(');

    if (node.properties.isEmpty) {
      buffer.write(')');
      return;
    }

    buffer.writeln();
    final entries = node.properties.entries.toList();
    for (var i = 0; i < entries.length; i++) {
      final entry = entries[i];
      _writeIndent(buffer, indent + 1);
      buffer.write('${entry.key}: ');
      _emitValue(buffer, entry.value, indent: indent + 1);
      if (i < entries.length - 1) buffer.write(',');
      buffer.writeln();
    }
    _writeIndent(buffer, indent);
    buffer.write(')');
  }

  void _emitValue(StringBuffer buffer, IrValue value, {required int indent}) {
    switch (value) {
      case IrStringValue():
        buffer.write('"${_escapeString(value.value)}"');
      case IrNumberValue():
        buffer.write(_formatDouble(value.value));
      case IrIntValue():
        buffer.write('0x${value.value.toRadixString(16).toUpperCase().padLeft(8, '0')}');
      case IrBoolValue():
        buffer.write(value.value.toString());
      case IrEnumValue():
        buffer.write('"${value.value}"');
      case IrListValue():
        _emitList(buffer, value, indent: indent);
      case IrMapValue():
        _emitMap(buffer, value, indent: indent);
      case IrWidgetNode():
        _emitWidget(buffer, value, indent: indent);
    }
  }

  void _emitList(StringBuffer buffer, IrListValue list, {required int indent}) {
    // Short list (all primitives, <=4 items) → inline
    if (list.values.length <= 4 && list.values.every(_isPrimitive)) {
      buffer.write('[');
      for (var i = 0; i < list.values.length; i++) {
        if (i > 0) buffer.write(', ');
        _emitValue(buffer, list.values[i], indent: indent);
      }
      buffer.write(']');
      return;
    }

    // Long list or contains widgets → multi-line
    buffer.writeln('[');
    for (var i = 0; i < list.values.length; i++) {
      _writeIndent(buffer, indent + 1);
      _emitValue(buffer, list.values[i], indent: indent + 1);
      if (i < list.values.length - 1) buffer.write(',');
      buffer.writeln();
    }
    _writeIndent(buffer, indent);
    buffer.write(']');
  }

  void _emitMap(StringBuffer buffer, IrMapValue map, {required int indent}) {
    buffer.writeln('{');
    final entries = map.entries.entries.toList();
    for (var i = 0; i < entries.length; i++) {
      _writeIndent(buffer, indent + 1);
      buffer.write('${entries[i].key}: ');
      _emitValue(buffer, entries[i].value, indent: indent + 1);
      if (i < entries.length - 1) buffer.write(',');
      buffer.writeln();
    }
    _writeIndent(buffer, indent);
    buffer.write('}');
  }

  bool _isPrimitive(IrValue value) {
    return value is IrStringValue ||
        value is IrNumberValue ||
        value is IrIntValue ||
        value is IrBoolValue ||
        value is IrEnumValue;
  }

  void _writeIndent(StringBuffer buffer, int level) {
    buffer.write('  ' * level);
  }

  String _formatDouble(double value) {
    if (value == value.truncateToDouble()) {
      return '${value.toStringAsFixed(1)}';
    }
    return value.toString();
  }

  String _escapeString(String s) {
    return s.replaceAll('\\', '\\\\').replaceAll('"', '\\"');
  }
}

// packages/rfw_gen/lib/src/converter.dart
import 'dart:typed_data';

import 'package:analyzer/dart/analysis/utilities.dart';
import 'package:analyzer/dart/ast/ast.dart';
import 'package:rfw/formats.dart';

import 'ast_visitor.dart';
import 'expression_converter.dart';
import 'rfwtxt_emitter.dart';
import 'widget_registry.dart';

/// Converts Flutter widget code to rfwtxt format.
class RfwConverter {
  final WidgetRegistry registry;
  final WidgetAstVisitor _astVisitor;
  final RfwtxtEmitter _emitter;

  RfwConverter({required this.registry})
      : _astVisitor = WidgetAstVisitor(
          registry: registry,
          expressionConverter: ExpressionConverter(),
        ),
        _emitter = RfwtxtEmitter();

  /// Convert a Dart source string to rfwtxt.
  ///
  /// The source should contain a top-level function (optionally annotated
  /// with @RfwWidget) that returns a widget tree.
  String convertFromSource(String dartSource) {
    // Wrap in a minimal compilation unit if needed
    final source = _ensureCompilationUnit(dartSource);
    final parseResult = parseString(content: source);
    final unit = parseResult.unit;

    // Find the first function declaration
    final function = unit.declarations
        .whereType<FunctionDeclaration>()
        .firstOrNull;

    if (function == null) {
      throw ArgumentError('No function declaration found in source');
    }

    return convertFromAst(function);
  }

  /// Convert a parsed AST function declaration to rfwtxt.
  String convertFromAst(FunctionDeclaration function) {
    final widgetName = _extractWidgetName(function);
    final ir = _astVisitor.extractWidgetTree(function);

    // Collect required imports from the widget tree
    final imports = _collectImports(ir);

    return _emitter.emit(
      widgetName: widgetName,
      root: ir,
      imports: imports,
    );
  }

  /// Convert rfwtxt string to RFW binary blob.
  Uint8List toBlob(String rfwtxt) {
    final library = parseLibraryFile(rfwtxt);
    return encodeLibraryBlob(library);
  }

  /// Extract widget name from @RfwWidget annotation or function name.
  String _extractWidgetName(FunctionDeclaration function) {
    // Check for @RfwWidget annotation
    for (final annotation in function.metadata) {
      if (annotation.name.name == 'RfwWidget') {
        final args = annotation.arguments;
        if (args != null && args.arguments.isNotEmpty) {
          final first = args.arguments.first;
          if (first is StringLiteral) {
            return first.stringValue ?? function.name.lexeme;
          }
        }
      }
    }

    // Fall back to function name, strip 'build' prefix
    final name = function.name.lexeme;
    if (name.startsWith('build') && name.length > 5) {
      return name[5].toLowerCase() + name.substring(6);
    }
    return name;
  }

  /// Collect required rfwtxt imports from the IR tree.
  /// v0.1: all 5 widgets are core widgets, so always returns {'core.widgets'}.
  /// Future versions will walk the IR tree to detect material widget usage.
  Set<String> _collectImports(IrWidgetNode root) {
    // TODO(v0.4): walk tree to detect material widgets and add 'material' import
    return {'core.widgets'};
  }

  /// Ensure the source is a valid Dart compilation unit.
  String _ensureCompilationUnit(String source) {
    // If source doesn't have imports for types used, add them
    // For now, just return as-is — analyzer can parse without imports
    // for syntactic analysis
    return source;
  }
}

// packages/rfw_gen_builder/lib/src/rfw_widget_builder.dart
import 'dart:async';

import 'package:analyzer/dart/ast/ast.dart';
import 'package:build/build.dart';
import 'package:analyzer/dart/analysis/utilities.dart';
import 'package:rfw_gen/rfw_gen.dart';

/// Custom Builder that finds @RfwWidget-annotated functions
/// and generates .rfwtxt and .rfw files.
class RfwWidgetBuilder implements Builder {
  final BuilderOptions options;

  RfwWidgetBuilder(this.options);

  @override
  Map<String, List<String>> get buildExtensions => {
        '.dart': ['.rfwtxt', '.rfw'],
      };

  @override
  FutureOr<void> build(BuildStep buildStep) async {
    final inputId = buildStep.inputId;
    final source = await buildStep.readAsString(inputId);

    // Parse the source
    final parseResult = parseString(content: source);
    final unit = parseResult.unit;

    // Find @RfwWidget-annotated functions
    final annotatedFunctions = <FunctionDeclaration>[];
    for (final declaration in unit.declarations) {
      if (declaration is FunctionDeclaration) {
        for (final annotation in declaration.metadata) {
          if (annotation.name.name == 'RfwWidget') {
            annotatedFunctions.add(declaration);
          }
        }
      }
    }

    if (annotatedFunctions.isEmpty) return;

    // Convert each annotated function
    final converter = RfwConverter(registry: WidgetRegistry.core());
    final rfwtxtParts = <String>[];

    for (final function in annotatedFunctions) {
      try {
        final rfwtxt = converter.convertFromAst(function);
        rfwtxtParts.add(rfwtxt);
      } catch (e) {
        log.severe('Failed to convert ${function.name.lexeme}: $e');
      }
    }

    if (rfwtxtParts.isEmpty) return;

    final combinedRfwtxt = rfwtxtParts.join('\n');

    // Write .rfwtxt
    final rfwtxtId = inputId.changeExtension('.rfwtxt');
    await buildStep.writeAsString(rfwtxtId, combinedRfwtxt);

    // Write .rfw binary
    try {
      final blob = converter.toBlob(combinedRfwtxt);
      final rfwId = inputId.changeExtension('.rfw');
      await buildStep.writeAsBytes(rfwId, blob);
    } catch (e) {
      log.warning('Failed to encode binary: $e');
    }
  }
}
```
